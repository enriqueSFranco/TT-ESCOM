\begin{UseCase}{GRAL-CU2}{Registrar Cuenta.}{
	Permite al actor registrarse como usuario dentro del sistema para poder consultar y/o aplicar a vacantes registradas en el sistema.  
}
	%----------------------------------------------------------------
	% Datos generales del CU:
	\UCsection{Atributos}
	\UCitem{Actor(es)}{
		Alumno. 
	}
	\UCitem[admin]{Prioridad}{
		Alta
	}
	\UCitem[admin]{Complejidad}{
		Media
	}
	\UCitem{Precondiciones}{
		\begin{Titemize}
			\Titem El número de boleta del alumno no debe estar asociado a una cuenta dentro del sistema.
			\Titem El correo del usuario no debe de estar asociado a otro usuario dentro del sistema.
		\end{Titemize}	
	}
	\UCitem{Postcondiciones}{
		El actor podrá ingresar al sistema y consultar o aplicar a las vacantes de su preferencia.
	}
	\UCitems{Entradas}{
	}
	\UCitems{Salidas}{
	}   
	\UCitem{Destino}{
		GRAL-USR-IU1.
	}

	%}

	%	\UCerr{Dos}{Los datos ingresados no existen dentro del sistema, el sistema indica el error con el mensaje \refIdElem{MSG002} en los campos correspondientes,}{ regresa al paso \ref{cu3-dd} de la trayectoria principal.}
	%}
	\UCitem{Viene de}{
		\ref	Elem{GRAL-CU1}.
	}	
\end{UseCase}

%Trayectoria Principal
\begin{UCtrayectoria}
	\UCpaso [\UCactor] Solicita igresar al sitio web de la \textbf{Bolsa de trabajo de ESCOM} desde el navegador de su preferencia.
    \UCpaso [\UCsist] Muestra la pantalla GRAL-USR-IU1.
	\UCpaso [\UCactor] Da clic en el botón \IUbutton{Alumno}. \refTray{A}
	\UCpaso [\UCsist] Muestra la pantalla GRAL-USR-IU1-1.
	\UCpaso [\UCactor] Introduce los datos para identificarse ante el sistema que son usuario y contraseña.\label{cu3-dd} \refErr{Uno}
	\UCpaso [\UCactor] Da clic en el botón \IUbutton{Iniciar Sesión}. 
    \UCpaso [\UCsist] Valida que el nombre de usuario y la contraseña ingresados esten registrados en el sistema. \refErr{Dos}
    \UCpaso [\UCsist] Muestra la pantalla GRAL-USR-IU3a.
\end{UCtrayectoria}
