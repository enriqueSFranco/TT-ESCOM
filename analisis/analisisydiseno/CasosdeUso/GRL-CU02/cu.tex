\begin{UseCase}[]{GRL-CU02}{Enviar Pre-Registro.}{
	Permite al actor registrarse como en el sistema y así posteriormente gestionar el perfil de la empresa, 
	publicar  vacantes, consultar perfiles de candidatos y gestionar sus solicitudes.
}
	%----------------------------------------------------------------
	% Datos generales del CU:
	\UCsection{Atributos}
	\UCitem{Actor(es)}{
		Reglutador. 
	}
	\UCitem[admin]{Prioridad}{
		Alta
	}
	\UCitem[admin]{Complejidad}{
		Alta
	}
	\UCitem{Precondiciones}{
		\Titem El correo electrónico del reclutador no debe estar registrado en el sistema.
	}
	\UCitem{Postcondiciones}{
		El actor podra gestionar su perfil, publicar vacantes y gestionar sus solicitudes.
	}
	\UCitems{Entradas}{
		\imprimeUC{entrada}
	}
	\UCitems{Salidas}{
		\imprimeUC{salida}
	}   
	\UCitem{Destino}{
	}
	\UCitem{Reglas de Negocio}{
		
	}
	\UCitem{Viene de}{
		%Caso de uso primario.
	}	
\end{UseCase}

%Trayectoria Principal
\begin{UCtrayectoria}
	\UCpaso [\UCactor] Presiona el botón \textbf{Crear Cuenta} desde la pantalla GRL-IU04.
    \UCpaso [\UCsist] Muestra la pantalla GRL-IU05.
	\UCpaso [\UCactor] Ingresa los datos de la empresa: Nombre de la empresa, RFC, razon social, documento del SAT de la empresa.\label{cu02-grl1}\refTray{A}
	\UCpaso [\UCactor] Ingresa los datos de contacto:Nombre, apellidos, correo electrónico empresarial y un telefono de contacto.
	\UCpaso [\UCactor] Acepta los terminos de privacidad y condiciones del sistema.
	\UCpaso [\UCactor] Da clic en el botón \IUbutton{Enviar Pre-Registro}. \refTray{B}
    \UCpaso [\UCsist] Verifica que los campos marcados como obligatorios hayan sido ingresados de acuerdo a la
	regla de negocio \refIdElem{RN-N001}.\refTray{C}
	\UCpaso [\UCsist] Verifica que el RFC ingresado cumpla con el formato correcto de acuerdo a las reglas de negocio  \refIdElem{RN-N006}, \refIdElem{RN-N003}.\refTray{D}
	\UCpaso [\UCsist] Verifica que el RFC ingresado no coincida con otro registrado en el sistema de acuerdo a la regla de negocio  \refIdElem{RN-N005}.\refTray{E}
	\UCpaso [\UCsist] Verifica que el correo electronico ingresado cumpla con el formato correcto de acuerdo a las reglas de negocio  \refIdElem{RN-N002} y \refIdElem{RN-N003} .\refTray{D} \label{cu02-grl2}
	\UCpaso [\UCsist] Verifica que el correo electronico ingresado no coincida con otro registrado en el sistema deacuedo a la regla de negocio  \refIdElem{RN-N005}.\refTray{E}
	\UCpaso [\UCsist] Envia una notificación al encargado y colaboradores del sistema indicando que hay una nieva empresa por validar.
    \UCpaso [\UCsist] Muestra la pantalla GRL-IU04.
\end{UCtrayectoria}

	\UCpaso [\UCactor] Ingresa los datos de la empresa:Nombre de la empresa.
	\UCpaso [\UCactor] Ingresa los datos de contacto: Nombre, apellidos, correo electrónico empresarial y un telefono de contacto.
	\UCpaso [\UCactor] Acepta los terminos de privacidad y condiciones del sistema.
	\UCpaso [\UCactor] Da clic en el botón \IUbutton{Enviar Pre-Registro}.
