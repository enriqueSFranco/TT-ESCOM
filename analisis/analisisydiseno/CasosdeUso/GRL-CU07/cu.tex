\begin{UseCase}[]{GRL-CU07}{Crear perfil}{
	Permite al alumno subir por prumera vez información general de su perfil al sistema.
}
	%----------------------------------------------------------------
	% Datos generales del CU:
	\UCsection{Atributos}
	\UCitem{Actor(es)}{
		estudiante. 
	}
	\UCitem[admin]{Prioridad}{ 
		Alta
	}
	\UCitem[admin]{Complejidad}{
		Alta
	}
	\UCitem{Precondiciones}{
		\Titem El alumno debe de tener una cuenta en el sistema.
		\Titem Debe de ser la primera vez que el alumno inicie sesion en el sistema.
	}
	\UCitems{Entradas}{
		\imprimeUC{entrada}
	}
	\UCitem{Destino}{
		\Titem \refElem{GRL-IU07}
	}
	\UCitem{Reglas de Negocio}{
		\Titem \refIdElem{RN-N001}
		
	}
	\UCitem{Viene de}{
		Caso de uso \refIdElem{GRL-CU03}.
	}	
\end{UseCase}

%Trayectoria Principal
\begin{UCtrayectoria}
	\UCpaso [\UCactor] Presiona inisia sesion exitosamente desde la pantalla \refElem{GRL-IU03}.
    \UCpaso [\UCsist] Valida que sea su primer inicio de sesión despues de haber creado su cuenta.\refTray{A}
	\UCpaso [\UCsist] Muestra la interfaz \refElem{GRL-IU07}.
	\UCpaso [\UCactor] Ingresa el \entrada[nombre]{estudiante.nombre}, \entrada[nombre]{estudiante.apellidos}, \entrada[ubicación]{estudiante.residencia}, 
	indica si tiene \entrada[disponibilidad para viajar]{estudiante.viajar} e ingresa su \entrada[telefono]{estudiante.telefono}.\label{cu07-grl1}
	\UCpaso [\UCactor] Da clic en el botón \IUbutton{Siguiente}.
	\UCpaso [\UCsist] Verifica que los campos marcados como obligatorios hayan sido ingresados de acuerdo a la
	regla de negocio \refIdElem{RN-N001}.\refTray{B}
	\UCpaso [\UCsist] Verifica que el telefono ingresado cumpla con el formato correcto de acuerdo a la regla de negocio  \refIdElem{RN-N007}.  \refTray{C}
	\UCpaso [\UCsist] Muestra la interfaz \refElem{GRL-IU07-1}.
	\UCpaso [\UCactor] Ingresa la \entrada[carrera]{hacademico.carrera}, selecciona la \entrada[institución]{hacademico.intitucion} y selecciona
	la fecha en la que \entrada[inicio]{hacademico.inicio} y \entrada[final]{hacademico.fin} la carrera.
	\UCpaso [\UCactor] Da clic en el botón \IUbutton{Siguiente}.
	\UCpaso [\UCsist] Verifica que los campos marcados como obligatorios hayan sido ingresados de acuerdo a la
	regla de negocio \refIdElem{RN-N001}.\refTray{D}\label{cu07-grl2} 
	\UCpaso [\UCsist] Muestra la interfaz \refElem{GRL-IU07-2}.
	\UCpaso [\UCactor] Ingresa el  \entrada[puesto deseado]{estudiante.puesto-deseado}, el \entrada[salario deseado]{estudiante.salario} y selecciona
	el  \entrada[tipo de trabajo]{estudiante.modalidad} deseado.
	\UCpaso [\UCactor] Da clic en el botón \IUbutton{Siguiente}.
	\UCpaso [\UCsist] Verifica que los campos marcados como obligatorios hayan sido ingresados de acuerdo a la
	regla de negocio \refIdElem{RN-N001}.\refTray{E}\label{cu07-grl3} 
	\UCpaso [\UCsist] Muestra la interfaz \refElem{GRL-IU07-3}.
	\UCpaso [\UCactor] Selecciona almenos una \entrada[hard skill]{habilidad.descripcion} y soft skill.
	\UCpaso [\UCactor] Da clic en el botón \IUbutton{Finalizar}.
	\UCpaso [\UCsist] Verifica que los campos marcados como obligatorios hayan sido ingresados de acuerdo a la
	regla de negocio \refIdElem{RN-N001}.\refTray{F}\label{cu07-grl4} 
	\UCpaso [\UCsist] Muestra la pantalla \refElem{GRL-IU06}. %\refTray{G} \refTray{H}
\end{UCtrayectoria}
