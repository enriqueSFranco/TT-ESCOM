\begin{UseCase}[]{USR-CU02}{Consultar perfil}{
	Permite al candidato consultar la información de su perfil y darle la oportunidad de editar dicha información o agregar más datos a su perfil.
}
	%----------------------------------------------------------------
	% Datos generales del CU:
	\UCsection{Atributos}
	\UCitem{Actor(es)}{
		Candidato. 
	}
	\UCitem[admin]{Prioridad}{ 
		Alta
	}
	\UCitem[admin]{Complejidad}{
		Media
	}
	\UCitem{Precondiciones}{
		\Titem El candidato debe de tener una cuenta en el sistema.
	}
	\UCitem{Destino}{
		\Titem \refElem{USR-IU02}
	}
	\UCitem{Reglas de Negocio}{
		\Titem Ninguna
		
	}
	\UCitem{Viene de}{
		Caso de uso \refIdElem{GRL-CU03}.
	}	
\end{UseCase}

%Trayectoria Principal
\begin{UCtrayectoria}
	\UCpaso [\UCactor] Inicia sesión exitosamente desde la pantalla \refElem{GRL-IU03}.
    \UCpaso [\UCsist] Valida que no sea su primer inicio de sesión después de haber creado su cuenta.\refTray{A}
	\UCpaso [\UCsist] Muestra la interfaz \refElem{USR-IU02}.
	\UCpaso [\UCactor] Consulta la información registrada de su perfil. 
\end{UCtrayectoria}

\begin{UCtrayectoriaA}[Fin del caso de uso]{A}{El actor desea editar datos personales}
	\UCpaso [\UCsist] Presiona el ícono \IUEditar{} desde la interfaz \refElem{USR-IU02} desde su sección correspondiente.
	\extendUC{USR-CU02.1}.
\end{UCtrayectoriaA} 

\begin{UCtrayectoriaA}[Fin del caso de uso]{B}{El actor desea editar objetivos profesionales}
	\UCpaso [\UCsist] Presiona el ícono \IUEditar{} desde la interfaz \refElem{USR-IU02} desde su sección correspondiente.
	\extendUC{USR-CU02.4}.
\end{UCtrayectoriaA} 

\begin{UCtrayectoriaA}[Fin del caso de uso]{C}{El actor desea editar medios de contacto}
	\UCpaso [\UCsist] Presiona el ícono \IUEditar{} desde la interfaz \refElem{USR-IU02} desde su sección correspondiente.
	\extendUC{USR-CU02.2}.
\end{UCtrayectoriaA} 

\begin{UCtrayectoriaA}[Fin del caso de uso]{D}{El actor desea editar habilidades}
	\UCpaso [\UCsist] Presiona el ícono \IUEditar{} desde la interfaz \refElem{USR-IU02} desde su sección correspondiente.
	\extendUC{USR-CU02.3}.
\end{UCtrayectoriaA} 

\begin{UCtrayectoriaA}[Fin del caso de uso]{F}{El actor desea consultar la sección ``Idiomas''}
	\UCpaso [\UCsist] Presiona el botón \IUbutton{Idiomas} desde la interfaz \refElem{USR-IU02}.
	\extendUC{USR-CU03}.
\end{UCtrayectoriaA}
