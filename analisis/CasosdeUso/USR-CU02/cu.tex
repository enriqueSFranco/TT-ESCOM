\begin{UseCase}[]{USR-CU02}{Consultar perfil}{
	Permite al alumno consultar la información de su perfil y darle la oportunidad de editar dicha información o agregar más datos a su perfil.
}
	%----------------------------------------------------------------
	% Datos generales del CU:
	\UCsection{Atributos}
	\UCitem{Actor(es)}{
		Candidato. 
	}
	\UCitem[admin]{Prioridad}{ 
		Alta
	}
	\UCitem[admin]{Complejidad}{
		Media
	}
	\UCitem{Precondiciones}{
		\Titem El alumno debe de tener una cuenta en el sistema.
	}
	\UCitem{Destino}{
		\Titem \refElem{USR-IU02}
	}
	\UCitem{Reglas de Negocio}{
		\Titem Ninguna
		
	}
	\UCitem{Viene de}{
		Caso de uso \refIdElem{GRL-CU03}.
	}	
\end{UseCase}

%Trayectoria Principal
\begin{UCtrayectoria}
	\UCpaso [\UCactor] Inicia sesión exitosamente desde la pantalla \refElem{GRL-IU03}.
    \UCpaso [\UCsist] Valida que no sea su primer inicio de sesión después de haber creado su cuenta.\refTray{A}
	\UCpaso [\UCsist] Muestra la interfaz \refElem{USR-IU02}.
	\UCpaso [\UCactor] Consulta la información registrada de su perfil. 
\end{UCtrayectoria}
