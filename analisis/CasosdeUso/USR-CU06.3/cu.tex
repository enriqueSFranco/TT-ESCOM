\begin{UseCase}[]{USR-CU06.3}{Eliminar historial academico}{
	Permite al candidato eliminar un hisotrial academico de su perfil.
}
	%----------------------------------------------------------------
	% Datos generales del CU:
	\UCsection{Atributos}
	\UCitem{Actor(es)}{
		Candidato. 
	}
	\UCitem[admin]{Prioridad}{ 
		Alta.
	}
	\UCitem[admin]{Complejidad}{
		Media.
	}
	\UCitem{Precondiciones}{
		El candidato debe de tener una cuenta en el sistema.
	}
	\UCitem{Destino}{
		\refElem{USR-IU04.3}
	}
	\UCitem{Viene de}{
		Caso de uso \refIdElem{GRL-CU04}.
	}	
\end{UseCase}

%Trayectoria Principal
\begin{UCtrayectoria}
	\UCpaso [\UCactor] Presiona el ícono \IUEliminar{} en la interfaz \refIdElem{USR-IU04} del historial académico que desea eliminar.
	\UCpaso Muestra la interfaz \refIdElem{USR-IU04.3}.
	\UCpaso [\UCsist] Presiona el botón \IUbutton{Aceptar}.\refTray{A}
	\UCpaso Elimina el registro en de la base de datos del sistema.
	\UCpaso Muestra la información actualizada en la interfaz \refIdElem{USR-IU04}. 
\end{UCtrayectoria}

\begin{UCtrayectoriaA}[Fin del caso de uso]{A}{El actor no desea eliminar la experiencia}
	\UCpaso [\UCsist] Presiona el botón \IUbutton{Cancelar}.
	\extendUC{USR-CU02}.
\end{UCtrayectoriaA}
