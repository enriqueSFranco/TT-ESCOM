\begin{UseCase}[]{USR-CU01}{Crear perfil}{
	Permite al alumno subir por primera vez su información general de su perfil al sistema y así 
	poder permitirle que aplique a las vacantes de su elección.
}
	%----------------------------------------------------------------
	% Datos generales del CU:
	\UCsection{Atributos}
	\UCitem{Actor(es)}{
		Candidato. 
	}
	\UCitem[admin]{Prioridad}{ 
		Alta
	}
	\UCitem[admin]{Complejidad}{
		Alta
	}
	\UCitem{Precondiciones}{
		\Titem El alumno debe de tener una cuenta en el sistema.
		\Titem Debe de ser la primera vez que el alumno inicie sesión en el sistema.
	}
	\UCitem{Destino}{
		\Titem \refElem{USR-IU01}
	}
	\UCitem{Reglas de Negocio}{
		\Titem \refIdElem{RN-N001}
		
	}
	\UCitem{Viene de}{
		Caso de uso \refIdElem{GRL-CU03}.
	}	
\end{UseCase}

%Trayectoria Principal
\begin{UCtrayectoria}
	\UCpaso [\UCactor] Presiona inicia sesión exitosamente desde la pantalla \refElem{GRL-IU03}.
    \UCpaso [\UCsist] Valida que sea su primer inicio de sesión después de haber creado su cuenta.\refTray{A}
	\UCpaso [\UCsist] Muestra la interfaz \refElem{USR-IU01}.
	\UCpaso [\UCactor] Ingresa el nombre, primer apellido, segundo apellido,ubicación, 
	indica si tiene disponibilidad para viajar e ingresa su teléfono.\label{USRCU01-grl1}
	\UCpaso [\UCactor] Da clic en el botón \IUbutton{Siguiente}.
	\UCpaso [\UCsist] Verifica que los campos marcados como obligatorios hayan sido ingresados de acuerdo a la
	regla de negocio \refIdElem{RN-N001}.\refTray{B}
	\UCpaso [\UCsist] Verifica que el teléfono ingresado cumpla con el formato correcto de acuerdo a la regla de negocio  \refIdElem{RN-N007}.  \refTray{C}
	\UCpaso [\UCsist] Muestra la interfaz \refElem{USR-IU01-1}.
	\UCpaso [\UCactor] Ingresa la carrera, selecciona la institución y selecciona
	la fecha en la que inicio y final la carrera.
	\UCpaso [\UCactor] Da clic en el botón \IUbutton{Siguiente}.
	\UCpaso [\UCsist] Verifica que los campos marcados como obligatorios hayan sido ingresados de acuerdo a la
	regla de negocio \refIdElem{RN-N001}.\refTray{D}\label{USRCU01-grl2} 
	\UCpaso [\UCsist] Muestra la interfaz \refElem{USR-IU01-2}.
	\UCpaso [\UCactor] Ingresa el  puesto deseado, el salario deseado y selecciona
	el  tipo de trabajo deseado.
	\UCpaso [\UCactor] Da clic en el botón \IUbutton{Siguiente}.
	\UCpaso [\UCsist] Verifica que los campos marcados como obligatorios hayan sido ingresados de acuerdo a la
	regla de negocio \refIdElem{RN-N001}.\refTray{E}\label{USRCU01-grl3} 
	\UCpaso [\UCsist] Muestra la interfaz \refElem{USR-IU01-3}.
	\UCpaso [\UCactor] Selecciona al menos una hard skill y soft skill.
	\UCpaso [\UCactor] Da clic en el botón \IUbutton{Finalizar}.
	\UCpaso [\UCsist] Verifica que los campos marcados como obligatorios hayan sido ingresados de acuerdo a la
	regla de negocio \refIdElem{RN-N001}.\refTray{F}\label{USRCU01-grl4} 
	\UCpaso [\UCsist] Muestra la pantalla \refElem{VCT-IU01}. %\refTray{G} \refTray{H}
\end{UCtrayectoria}

\begin{UCtrayectoriaA}[Fin del caso de uso]{A}{No es la primera vez que alctor inicia sesion.}
	\UCpaso [\UCsist] Muestra la interfaz \refIdElem{VCT-IU01}.
	\extendUC{VCT-CU01}.
\end{UCtrayectoriaA}
