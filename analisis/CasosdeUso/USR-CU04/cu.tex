\begin{UseCase}[]{USR-CU04}{Consultar historial académico}{
	Permite al alumno consultar la información de sus historial académico registrados en el perfil.
}
	%----------------------------------------------------------------
	% Datos generales del CU:
	\UCsection{Atributos}
	\UCitem{Actor(es)}{
		Candidato. 
	}
	\UCitem[admin]{Prioridad}{ 
		Alta
	}
	\UCitem[admin]{Complejidad}{
		Media
	}
	\UCitem{Precondiciones}{
		El alumno debe de tener una cuenta en el sistema.
	}
	\UCitem{Destino}{
		\refElem{USR-IU04}
	}
	\UCitem{Viene de}{
		Caso de uso \refIdElem{USR-CU01}.
	}	
\end{UseCase}

%Trayectoria Principal
\begin{UCtrayectoria}
	\UCpaso [\UCactor] Presiona el botón \IUbutton{historial académico} desde la interfaz \refElem{USR-IU02}.
    \UCpaso [\UCsist] Obtiene el nombre de la carrera, la institución, nivel académico, estado académico, fecha de inicio y la fecha en la que se planea terminar o terminó la carrera. de cada una que está registrada en el sistema.
	\UCpaso [\UCsist] Muestra la interfaz \refElem{USR-IU04}.\refTray{A}\refTray{B}\refTray{C}
\end{UCtrayectoria}

\begin{UCtrayectoriaA}[Fin del caso de uso]{A}{El actor desea registrar una certificación}
	\UCpaso [\UCsist] Presiona el ícono \IUAgregar{} desde la interfaz \refElem{USR-IU04} desde su sección correspondiente.
	\extendUC{USR-CU04.1}.
\end{UCtrayectoriaA} 

\begin{UCtrayectoriaA}[Fin del caso de uso]{A}{El actor desea editar una certificación}
	\UCpaso [\UCsist] Presiona el ícono \IUEditar{} desde la interfaz \refElem{USR-IU04} desde su sección correspondiente.
	\extendUC{USR-CU04.2}.
\end{UCtrayectoriaA} 

\begin{UCtrayectoriaA}[Fin del caso de uso]{A}{El actor desea eliminar una certificación}
	\UCpaso [\UCsist] Presiona el ícono \IUEliminar{} desde la interfaz \refElem{USR-IU04} desde su sección correspondiente.
	\extendUC{USR-CU04.3}.
\end{UCtrayectoriaA}
